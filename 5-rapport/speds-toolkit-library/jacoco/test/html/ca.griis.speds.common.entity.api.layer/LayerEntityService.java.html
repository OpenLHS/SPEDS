public class LayerEntityService implements EntityService {
  private static final GriisLogger logger = GriisLoggerFactory.getLogger(LayerEntityService.class);

  private final LoDadClientService loDadService;
  private final CommonHttpClient commonHttpClient;

  public LayerEntityService(LoDadClientService loDadService,
      SecurityCertificateVerification checkKeyServiceCertificate) throws Exception {
    this.loDadService = loDadService;
    commonHttpClient = new CommonHttpClient(checkKeyServiceCertificate);
  }

  @Override
  public Optional<PuAk> getPublicKey(PgaId pgaId, EntityCode entityCode) {
    logger.trace(Trace.ENTER_METHOD_2, "pgaId", pgaId, "entityCode", entityCode);
    Optional<PuAk> puAk = Optional.empty();

    // Récupération de entité représentant l'unité administrative des clés du PGA
    String iriStr = getKeyAdministrationEntity(pgaId);

    // Appel de sclé et récupération de la clé publique
    try {
      StringBuilder addr = new StringBuilder(iriStr);
      addr.append("/key").append("/").append(entityCode.code());
      HttpResponse<String> response = commonHttpClient.get(new URI(addr.toString()));
      if (response.statusCode() == HttpURLConnection.HTTP_OK) {
        MessageResponse<?> message;
        try {
          message = SharedObjectMapper.getInstance().getMapper().readValue(response.body(),
              new TypeReference<MessageResponse<RetrievedKey>>() {});
        } catch (InvalidFormatException ex) {
          message = SharedObjectMapper.getInstance().getMapper().readValue(response.body(),
              new TypeReference<MessageResponse<String>>() {});
          logger.info(Info.VARIABLE_LOGGING_1, "Unsuccessful to retrieve key: ",
              message.exceptionMsg());
        }
        if (message.communicationStatus() == CommunicationStatus.ST_OK) {
          String tmpKey = ((RetrievedKey) message.body()).value();
          puAk = Optional.of(new PuAk(Base64.getDecoder().decode(tmpKey)));
        }
      }
    } catch (IOException | InterruptedException | URISyntaxException e) {
      throw new EntityServiceException(e);
    }

    logger.trace(Trace.EXIT_METHOD_1, "puAk", puAk);
    return puAk;
  }

  @Override
  public Boolean verifyLegitimacy(PgaId pgaId, EntityCode entityCode, PuAk puAk) {
    logger.trace(Trace.ENTER_METHOD_3, "pgaId", pgaId, "entityCode", entityCode, "puAk", puAk);

    Optional<PuAk> key = getPublicKey(pgaId, entityCode);
    Boolean isVerified = key.isPresent() && Arrays.equals(key.get().puAk(), puAk.puAk());

    logger.trace(Trace.EXIT_METHOD_1, "isVerified", isVerified);
    return isVerified;
  }

  @Override
  public Optional<Entity> getEntity(PgaId pgaId, EntityCode entityCode) {
    logger.trace(Trace.ENTER_METHOD_2, "pgaId", pgaId, "entityCode", entityCode);
    Optional<Entity> entity = Optional.empty();

    EnvironmentService env = loDadService.getProjectService()
        .getEnvironment(new ca.griis.dadagem.type.lexicon.Guid(pgaId.pgaId()));

    List<EntityDef> entitiesDef = new ArrayList<>();
    entitiesDef.add(env.getEnv().getPlanEntities().getEeReferenceDef());
    entitiesDef.addAll(env.getEnv().getPlanEntities().getCeReferencesDef());
    entitiesDef.addAll(env.getEnv().getPlanEntities().getPeReferencesDef());

    entitiesDef.addAll(env.getEnv().getAdministrationEntities().getAesReferencesDef());
    entitiesDef.add(env.getEnv().getAdministrationEntities().getAecReferenceDef());
    entitiesDef.add(env.getEnv().getAdministrationEntities().getAekReferenceDef());

    Optional<EntityDef> aej = env.getEnv().getAdministrationEntities().getAejReferenceDef();
    if (aej.isPresent()) {
      entitiesDef.add(aej.get());
    }

    entitiesDef.addAll(env.getEnv().getAdministrationEntities().getAeReferencesDef());

    Optional<EntityDef> entityDef =
        entitiesDef.stream().filter(x -> x.getCode().equals(entityCode.code())).findFirst();
    if (entityDef.isPresent()) {
      String iriStr = entityDef.get().getProxyAddress();
      entity =
          Optional.of(new Entity(entityCode, IRIFactory.iriImplementation().construct(iriStr)));
    }

    logger.trace(Trace.EXIT_METHOD_1, "entity", entity);
    return entity;
  }

  private String getKeyAdministrationEntity(PgaId pgaId) {
    EnvironmentService env = loDadService.getProjectService()
        .getEnvironment(new ca.griis.dadagem.type.lexicon.Guid(pgaId.pgaId()));
    EntityDef scle = env.getEnv().getAdministrationEntities().getAekReferenceDef();
    String iriStr = scle.getProxyAddress();
    if (iriStr.equals("S.O.")) {
      iriStr = scle.getCode();
    }
    return iriStr;
  }
}
