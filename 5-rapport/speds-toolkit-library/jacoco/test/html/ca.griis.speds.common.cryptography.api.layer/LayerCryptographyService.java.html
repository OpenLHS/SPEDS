public class LayerCryptographyService implements CryptographyService {
  private static final GriisLogger logger = GriisLoggerFactory.getLogger(LayerEntityService.class);

  private final ConcurrentMap<Algorithm, AlgoLayerLink> algorithmForLayer;

  public LayerCryptographyService(ProtocolLayer layer) {
    logger.trace(Trace.ENTER_METHOD_1, "layer", layer);
    this.algorithmForLayer =
        ProtocolLayerAlgoDictionary.getInstance().getLayerAgorithmsGroup(layer);
  }

  @Override
  public SecretKey generateSecretKey() {
    logger.trace(Trace.ENTER_METHOD_0);
    AlgoLayerLink algoLink = algorithmForLayer.get(Algorithm.symmetricKeyGenerator);

    if (algoLink == null) {
      throw new CryptographyServiceException("Algorithms are unknown for this layer.");
    }

    SecretKeyGeneratorAlgorithm skga = null;
    switch (algoLink.format()) {
      case aes_gcm_nopadding:
      default:
        skga = SecretKeyGeneratorAlgorithm.AES;
        break;
    }
    AlgoLayerLink length = algorithmForLayer.get(Algorithm.symmetricKeyLength);
    String lengthStr = length.format().getFormat();
    SecretKey sk = SecretKeyGenerator.generateSymmetricKey(skga, Integer.decode(lengthStr));

    logger.trace(Trace.EXIT_METHOD_1, "secretKey", System.identityHashCode(sk));
    return sk;
  }

  @Override
  public KeyPair generateKeyPair(KeyPairAlgorithmType type) {
    logger.trace(Trace.ENTER_METHOD_1, "type", type);
    KeyPair keyPair = null;
    switch (type) {
      case signature:
        keyPair = generateSignatureKeyPair();
        break;
      case keyAgreement:
        keyPair = generateKeyAgreementKeyPair();
        break;
      case encryption:
      default: // @note [SSC] : default is needed for checkstyle but will never be called
        keyPair = generateAsymmetricEncryptionKeyPair();
        break;
    }
    logger.trace(Trace.EXIT_METHOD_1, "keyPair", System.identityHashCode(keyPair));
    return keyPair;
  }

  @Override
  public Hash hash(byte[] data) {
    logger.trace(Trace.ENTER_METHOD_1, "data", data);
    AlgoLayerLink algoLink = algorithmForLayer.get(Algorithm.hashing);

    if (algoLink == null) {
      throw new CryptographyServiceException("Algorithms are unknown for this layer.");
    }

    Hashing sha;
    switch (algoLink.format()) {
      case sha512:
      default:
        sha = new Sha512Hashing();
        break;
    }
    // Creation de l'empreinte numérique
    Hash stamp = sha.hash(data);
    logger.trace(Trace.EXIT_METHOD_1, "hash", stamp);
    return stamp;
  }

  @Override
  public Boolean verify(byte[] data, Hash hash) {
    logger.trace(Trace.ENTER_METHOD_2, "data", data, "hash", hash);
    Hash verify = hash(data);
    Boolean result = verify.equals(hash);
    logger.trace(Trace.EXIT_METHOD_1, "verify", result);
    return result;
  }

  @Override
  public Boolean verify(PublicKey publicKey, byte[] data, DigitalSignature signature) {
    logger.trace(Trace.ENTER_METHOD_3, "publicKey", publicKey, "data", data, "signature",
        signature);
    AlgoLayerLink algoLink = algorithmForLayer.get(Algorithm.signature);

    if (algoLink == null) {
      throw new CryptographyServiceException("Algorithms are unkwnown for this layer.");
    }

    VerifySigning verifySigning = null;
    switch (algoLink.format()) {
      case sha3_512_rsa:
      default:
        verifySigning = new RsaVerifySigning((RSAPublicKey) publicKey);
        break;
    }

    Boolean result = verifySigning.verify(data, signature);

    logger.trace(Trace.EXIT_METHOD_1, "verify", result);
    return result;
  }

  @Override
  public DigitalSignature sign(PrivateKey privateKey, byte[] data) {
    logger.trace(Trace.ENTER_METHOD_2, "privateKey", System.identityHashCode(privateKey), "data",
        data);
    AlgoLayerLink algoLink = algorithmForLayer.get(Algorithm.signature);

    if (algoLink == null) {
      throw new CryptographyServiceException("Algorithms are unknown for this layer.");
    }

    DigitalSignature ds = null;
    switch (algoLink.format()) {
      case sha3_512_rsa:
      default:
        RsaSigning rsaSign = new RsaSigning((RSAPrivateKey) privateKey);
        ds = rsaSign.sign(data);
    }
    logger.trace(Trace.EXIT_METHOD_1, "digitalSignature", ds);
    return ds;
  }

  @Override
  public byte[] symmetricEncrypt(SecretKey key, byte[] data) {
    logger.trace(Trace.ENTER_METHOD_2, "secretKey", System.identityHashCode(key), "data", data);
    AlgoLayerLink algoLink = algorithmForLayer.get(Algorithm.symmetricCipher);

    if (algoLink == null) {
      throw new CryptographyServiceException("Algorithms are unknown for this layer.");
    }

    Encryptor symmetricEncrypt = null;
    switch (algoLink.format()) {
      case aes_gcm_nopadding:
      default:
        symmetricEncrypt = new AesGcmEncryptor(key);
        break;
    }
    byte[] encryptedData = symmetricEncrypt.encrypt(data);
    logger.trace(Trace.EXIT_METHOD_1, "encryptedData", encryptedData);
    return encryptedData;
  }

  @Override
  public byte[] asymmetricEncrypt(PublicKey key, byte[] data) {
    logger.trace(Trace.ENTER_METHOD_2, "publicKey", key, "data", data);
    AlgoLayerLink algoLink = algorithmForLayer.get(Algorithm.asymmetricCipher);

    if (algoLink == null) {
      throw new CryptographyServiceException("Algorithms are unknown for this layer.");
    }

    Encryptor encryptor = null;
    switch (algoLink.format()) {
      case rsa:
      default:
        encryptor = new RsaEncryptor((RSAPublicKey) key);
        break;
    }
    byte[] encryptedData = encryptor.encrypt(data);
    logger.trace(Trace.EXIT_METHOD_1, "encryptedData", encryptedData);
    return encryptedData;
  }

  @Override
  public byte[] symmetricDecrypt(SecretKey key, byte[] encryptedData) {
    logger.trace(Trace.ENTER_METHOD_2, "secretKey", System.identityHashCode(key), "encryptedData",
        encryptedData);
    AlgoLayerLink algoLink = algorithmForLayer.get(Algorithm.symmetricCipher);

    if (algoLink == null) {
      throw new CryptographyServiceException("Algorithms are unknown for this layer.");
    }

    Decryptor decryptor = null;
    switch (algoLink.format()) {
      case aes_gcm_nopadding:
      default:
        decryptor = new AesGcmDecryptor(key);
        break;
    }

    try {
      byte[] decryptedContent = decryptor.decrypt(encryptedData);
      logger.trace(Trace.EXIT_METHOD_1, "decryptedContent",
          System.identityHashCode(decryptedContent));
      return decryptedContent;
    } catch (Exception e) {
      throw new DecryptException(e);
    }
  }

  @Override
  public byte[] asymmetricDecrypt(PrivateKey key, byte[] encryptedData) {
    logger.trace(Trace.ENTER_METHOD_2, "privateKey", System.identityHashCode(key), "encryptedData",
        encryptedData);
    AlgoLayerLink algoLink = algorithmForLayer.get(Algorithm.asymmetricCipher);

    if (algoLink == null) {
      throw new CryptographyServiceException("Algorithms are unknown for this layer.");
    }

    Decryptor decryptor = null;
    switch (algoLink.format()) {
      case rsa:
      default:
        decryptor = new RsaDecryptor((RSAPrivateKey) key);
        break;
    }

    try {
      byte[] decryptedData = decryptor.decrypt(encryptedData);
      logger.trace(Trace.EXIT_METHOD_1, "decrypteData", System.identityHashCode(decryptedData));
      return decryptedData;
    } catch (Exception e) {
      throw new DecryptException(e);
    }
  }

  @Override
  public SecretKey keyAgreement(KeyPair originatorKeyPair, PublicKey recipientPublicKey) {
    logger.trace(Trace.ENTER_METHOD_2, "originatorKeyPair",
        System.identityHashCode(originatorKeyPair), "recipientPublicKey",
        System.identityHashCode(recipientPublicKey));

    AlgoLayerLink algoLink = algorithmForLayer.get(Algorithm.keyAgreement);

    if (algoLink == null) {
      throw new CryptographyServiceException("Algorithms are unknown for this layer.");
    }

    try {
      KeyAgreementProvider keyAgreementProvider = null;
      switch (algoLink.format()) {
        case x25519:
        default:
          keyAgreementProvider = new X25519Provider();
          break;
      }

      keyAgreementProvider.initializeAgreement(originatorKeyPair.getPrivate());
      byte[] sharedSecretKey = keyAgreementProvider.completeAgreement(recipientPublicKey);

      byte[] derivedSecuredKey = keyAgreementProvider.deriveSecureKey(sharedSecretKey,
          originatorKeyPair.getPublic(), recipientPublicKey);
      SecretKey sk = new SecretKeySpec(derivedSecuredKey, 0, derivedSecuredKey.length,
          algorithmForLayer.get(Algorithm.symmetricKeyGenerator).format().getFormat());

      logger.trace(Trace.EXIT_METHOD_1, "secretKey", System.identityHashCode(sk));
      return sk;
    } catch (InvalidKeyException | IllegalStateException | NoSuchAlgorithmException e) {
      throw new CryptographyServiceException(e);
    }
  }

  protected KeyAgreementProvider getKeyAgreementX25519Provider() throws NoSuchAlgorithmException {
    return new X25519Provider();
  }

  /**
   * @brief @~english «Description of the function»
   * @param «parameter name» «Parameter description»
   * @exception «exception name» «Exception description»
   * @return «Return description»
   *
   * @brief @~french Génération d'une paire de clés pour KeyAgreement
   * @return la paire de clés générée
   */
  private KeyPair generateKeyAgreementKeyPair() {
    logger.trace(Trace.ENTER_METHOD_0);
    AlgoLayerLink algoLink = algorithmForLayer.get(Algorithm.keyAgreementKeyPairGenerator);

    if (algoLink == null) {
      throw new CryptographyServiceException("Algorithms are unknown for this layer.");
    }

    KeyAgreementProvider keyAgreementProvider;

    AlgorithmFormat format = algoLink.format();
    switch (format) {
      case x25519:
      default:
        try {
          keyAgreementProvider = getKeyAgreementX25519Provider();
        } catch (NoSuchAlgorithmException e) {
          throw new CryptographyServiceException(e);
        }
        break;
    }

    KeyPair keyPair = keyAgreementProvider.generateEphemeralKeys();
    logger.trace(Trace.EXIT_METHOD_1, "keyPair", System.identityHashCode(keyPair));
    return keyPair;
  }


  /**
   * @brief @~english «Description of the function»
   * @param «parameter name» «Parameter description»
   * @exception «exception name» «Exception description»
   * @return «Return description»
   *
   * @brief @~french Génération d'une paire de clés pour Signature
   * @return la paire de clés générée
   */
  private KeyPair generateSignatureKeyPair() {
    logger.trace(Trace.ENTER_METHOD_0);
    AlgoLayerLink algoLink = algorithmForLayer.get(Algorithm.signatureKeyPairGenerator);

    if (algoLink == null) {
      throw new CryptographyServiceException("Algorithms are unknown for this layer.");
    }

    KeyPair keyPair = null;
    AlgorithmFormat format = algoLink.format();
    switch (format) {
      case rsa:
      default:
        AlgoLayerLink length = algorithmForLayer.get(Algorithm.asymmetricKeyPairLength);
        String lengthStr = length.format().getFormat();
        RsaKeysGenerator keyGenerator = new RsaKeysGenerator(Integer.decode(lengthStr));
        keyPair = keyGenerator.generateKeyPair();
        break;
    }

    logger.trace(Trace.EXIT_METHOD_1, "keyPair", System.identityHashCode(keyPair));
    return keyPair;
  }

  /**
   * @brief @~english «Description of the function»
   * @param «parameter name» «Parameter description»
   * @exception «exception name» «Exception description»
   * @return «Return description»
   *
   * @brief @~french Génération d'une paire de clés pour Encryption asymmetrique
   * @return la paire de clés générée
   */
  private KeyPair generateAsymmetricEncryptionKeyPair() {
    logger.trace(Trace.ENTER_METHOD_0);
    AlgoLayerLink algoLink = algorithmForLayer.get(Algorithm.asymmetricKeyPairGenerator);

    if (algoLink == null) {
      throw new CryptographyServiceException("Algorithms are unknown for this layer.");
    }
    KeyPair keyPair = null;

    switch (algoLink.format()) {
      case rsa:
      default:
        AlgoLayerLink length = algorithmForLayer.get(Algorithm.asymmetricKeyPairLength);
        String lengthStr = length.format().getFormat();
        RsaKeysGenerator keyGenerator = new RsaKeysGenerator(Integer.decode(lengthStr));
        keyPair = keyGenerator.generateKeyPair();
        break;
    }
    logger.trace(Trace.EXIT_METHOD_1, "keyPair", System.identityHashCode(keyPair));
    return keyPair;
  }
}
