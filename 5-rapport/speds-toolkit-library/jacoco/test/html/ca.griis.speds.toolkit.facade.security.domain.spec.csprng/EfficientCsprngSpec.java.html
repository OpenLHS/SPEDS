package ca.griis.speds.toolkit.facade.security.domain.spec.csprng;

import java.nio.charset.StandardCharsets;
import java.security.DrbgParameters;
import java.security.SecureRandomParameters;

public class EfficientCsprngSpec extends CsprngSpec {
  private final SecureRandomParameters params;

  public EfficientCsprngSpec() {
    this.params = DrbgParameters.instantiation(
        256, DrbgParameters.Capability.PR_AND_RESEED,
        "StrongSecureRandom-v1".getBytes(StandardCharsets.UTF_8));
  }

  public SecureRandomParameters getParams() {
    return params;
  }
}
