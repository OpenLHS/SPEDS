package ca.griis.speds.toolkit.facade.security.domain.spec.generator.asym;

import java.security.spec.RSAKeyGenParameterSpec;

public class StrongRsaKeyPairGenSpec extends AsymKeyPairGenSpec {
  private final RSAKeyGenParameterSpec keyGenParameterSpec;

  public StrongRsaKeyPairGenSpec() {
    super("RSA");

    this.keyGenParameterSpec = new RSAKeyGenParameterSpec(4096, RSAKeyGenParameterSpec.F4);
  }

  public RSAKeyGenParameterSpec getKeyGenParameterSpec() {
    return keyGenParameterSpec;
  }
}
