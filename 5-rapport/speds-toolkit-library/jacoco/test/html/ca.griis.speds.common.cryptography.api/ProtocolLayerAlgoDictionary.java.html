package ca.griis.speds.common.cryptography.api;

import ca.griis.speds.common.cryptography.domain.AlgoLayerLink;
import ca.griis.speds.common.cryptography.domain.Algorithm;
import ca.griis.speds.common.cryptography.domain.AlgorithmFormat;
import ca.griis.speds.common.cryptography.domain.ProtocolLayer;
import java.util.Map;
import java.util.concurrent.ConcurrentHashMap;
import java.util.concurrent.ConcurrentMap;

/**
 * @brief @~english «Brief component description (class, interface, ...)»
 * @par Details
 *      «Detailed description of the component (optional)»
 * @par Model
 *      «Model (Abstract, automation, etc.) (optional)»

 * @par Limits
 *      «Limits description (optional)»
 *
 * @brief @~french Un singleton dénotant les différents algorithmes cryptographiques utilisés dans
 *        les différentes couches du protocole regroupé selon la couche et selon le type.
 * @par Details
 *      S.O.
 * @par Modèle
 *      S.O.
 * @par Conception
 *      S.O.
 * @par Limites
 *      S.O.
 *
 * @par Historique
 *      2025-04-09 [SSC] - Implémentation initiale<br>
 *
 * @par Tâches
 *      S.O.
 */
public class ProtocolLayerAlgoDictionary {

  private final Map<ProtocolLayer, ConcurrentMap<Algorithm, AlgoLayerLink>> layerAlgorithm;

  public static ProtocolLayerAlgoDictionary getInstance() {
    return LazyHolder.instance;
  }

  private static class LazyHolder {
    static final ProtocolLayerAlgoDictionary instance = new ProtocolLayerAlgoDictionary();
  }

  private ProtocolLayerAlgoDictionary() {
    ConcurrentMap<Algorithm, AlgoLayerLink> tmp = new ConcurrentHashMap<>();
    layerAlgorithm = new ConcurrentHashMap<>();
    // network
    tmp.put(Algorithm.asymmetricKeyPairLength, new AlgoLayerLink(Algorithm.asymmetricKeyPairLength,
        AlgorithmFormat.length4096, ProtocolLayer.network));
    tmp.put(Algorithm.signatureKeyPairGenerator, new AlgoLayerLink(
        Algorithm.asymmetricKeyPairGenerator, AlgorithmFormat.rsa, ProtocolLayer.network));
    tmp.put(Algorithm.signature, new AlgoLayerLink(Algorithm.signature,
        AlgorithmFormat.sha3_512_rsa, ProtocolLayer.network));
    tmp.put(Algorithm.hashing,
        new AlgoLayerLink(Algorithm.hashing, AlgorithmFormat.sha512, ProtocolLayer.network));

    layerAlgorithm.put(ProtocolLayer.network, tmp);
    tmp = new ConcurrentHashMap<>();

    // transport
    tmp.put(Algorithm.hashing,
        new AlgoLayerLink(Algorithm.hashing, AlgorithmFormat.sha512, ProtocolLayer.transport));

    layerAlgorithm.put(ProtocolLayer.transport, tmp);
    tmp = new ConcurrentHashMap<>();

    // session
    tmp.put(Algorithm.symmetricKeyLength, new AlgoLayerLink(Algorithm.symmetricKeyLength,
        AlgorithmFormat.length256, ProtocolLayer.session));
    tmp.put(Algorithm.symmetricKeyGenerator, new AlgoLayerLink(Algorithm.symmetricKeyGenerator,
        AlgorithmFormat.aes, ProtocolLayer.session));
    tmp.put(Algorithm.asymmetricKeyPairLength, new AlgoLayerLink(Algorithm.asymmetricKeyPairLength,
        AlgorithmFormat.length4096, ProtocolLayer.session));
    tmp.put(Algorithm.asymmetricKeyPairGenerator, new AlgoLayerLink(
        Algorithm.asymmetricKeyPairGenerator, AlgorithmFormat.rsa, ProtocolLayer.session));
    tmp.put(Algorithm.signatureKeyPairGenerator,
        new AlgoLayerLink(Algorithm.signature, AlgorithmFormat.rsa, ProtocolLayer.session));
    tmp.put(Algorithm.keyAgreementKeyPairGenerator, new AlgoLayerLink(
        Algorithm.keyAgreementKeyPairGenerator, AlgorithmFormat.x25519, ProtocolLayer.session));
    tmp.put(Algorithm.symmetricCipher, new AlgoLayerLink(Algorithm.symmetricCipher,
        AlgorithmFormat.aes_gcm_nopadding, ProtocolLayer.session));
    tmp.put(Algorithm.asymmetricCipher,
        new AlgoLayerLink(Algorithm.asymmetricCipher, AlgorithmFormat.rsa, ProtocolLayer.session));
    tmp.put(Algorithm.signature, new AlgoLayerLink(Algorithm.signature,
        AlgorithmFormat.sha3_512_rsa, ProtocolLayer.session));
    tmp.put(Algorithm.keyAgreement,
        new AlgoLayerLink(Algorithm.keyAgreement, AlgorithmFormat.x25519, ProtocolLayer.session));
    tmp.put(Algorithm.hashing,
        new AlgoLayerLink(Algorithm.hashing, AlgorithmFormat.sha512, ProtocolLayer.session));

    layerAlgorithm.put(ProtocolLayer.session, tmp);
    tmp = new ConcurrentHashMap<>();

    // Presentation
    tmp.put(Algorithm.symmetricKeyLength, new AlgoLayerLink(Algorithm.symmetricKeyLength,
        AlgorithmFormat.length256, ProtocolLayer.presentation));
    tmp.put(Algorithm.symmetricKeyGenerator, new AlgoLayerLink(Algorithm.symmetricKeyGenerator,
        AlgorithmFormat.aes, ProtocolLayer.presentation));
    tmp.put(Algorithm.symmetricCipher, new AlgoLayerLink(Algorithm.symmetricCipher,
        AlgorithmFormat.aes_gcm_nopadding, ProtocolLayer.presentation));

    layerAlgorithm.put(ProtocolLayer.presentation, tmp);
  }

  public ConcurrentMap<Algorithm, AlgoLayerLink> getLayerAgorithmsGroup(ProtocolLayer layer) {
    return layerAlgorithm.getOrDefault(layer, new ConcurrentHashMap<>());
  }
}
