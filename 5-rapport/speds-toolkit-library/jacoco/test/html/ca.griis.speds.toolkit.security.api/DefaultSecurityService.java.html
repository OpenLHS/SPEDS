import org.apache.jena.iri.IRI;

 *
 * @par Tâches
 *      S.O.
 */
public class DefaultSecurityService implements SecurityService {
  private final SecurityFacade facade;

  public DefaultSecurityService(SecurityFacade facade) {
    this.facade = facade;
  }

  @Override
  public Boolean validateCertificate(X509Certificate certificate, IRI iri) {
    Boolean isValid = facade.verifyCertificateTemporalValidity(certificate)
        && facade.verifyCertificateChain(certificate)
        && facade.verifyCertificateExtensions(certificate);

    return isValid;
  }
}
