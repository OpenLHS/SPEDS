package ca.griis.speds.toolkit.facade.security.domain.spec.dh;

public class StrongX25519KeyAgreementFnSpec extends X25519KeyAgreementFnSpec {
  private final String hashAlgo;
  private final Integer keyBitLength;

  public StrongX25519KeyAgreementFnSpec() {
    this.hashAlgo = "SHA-256";
    this.keyBitLength = 256;
  }

  public String getHashAlgo() {
    return hashAlgo;
  }

  public Integer getKeyBitLength() {
    return keyBitLength;
  }
}
