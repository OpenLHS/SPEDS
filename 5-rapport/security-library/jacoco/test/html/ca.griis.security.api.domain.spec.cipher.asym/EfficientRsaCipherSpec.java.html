public class EfficientRsaCipherSpec extends RsaCipherSpec {
  public EfficientRsaCipherSpec() {
    super(
        "RSA/ECB/OAEPWithSHA-256AndMGF1Padding",
        new OAEPParameterSpec(
            "SHA-256",
            "MGF1",
            MGF1ParameterSpec.SHA256,
            PSource.PSpecified.DEFAULT));
  }

  @Override
  public Map<String, String> getParameters() {
    return Map.of(
        "mdName", "SHA-256",
        "mgfName", "MGF1",
        "mgfSpec", "SHA-256",
        "pSrc", "");
  }
}
